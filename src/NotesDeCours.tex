\newcommand\Twocolor[4]{
\path 
	let 
	\p1=($ #1 - #2 $ ),
	\p2=#1,
	\p3=#2
	in 
	node[
		draw,
		outer sep=0pt,
		inner sep=0pt,
		minimum height=3pt,
		text width={veclen(\x1,\y1)},
		rotate={atan((\y3-\y2)/(\x3-\x2))},
		anchor=west
	] (mydouble) at #1 {};
	\draw[#3,line width=1pt] (mydouble.north west) -- (mydouble.north east); 
	\draw[#4,line width=1pt] (mydouble.south west) -- (mydouble.south east); 
}
